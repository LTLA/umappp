21inline Eigen::MatrixXd normalized_laplacian_by_component(const NeighborList& edges, const ComponentIndices& comp_info, int component, int ndim) {
22 const auto& which = comp_info.reversed[component];
23 const auto& indices = comp_info.new_indices;
24
25 std::vector<double> sums(which.size());
26 for (size_t c = 0; c < which.size(); ++c) {
27 const auto& current = edges[which[c]];
28 double& sum = sums[c];
29 for (const auto& f : current) {
30 sum += f.second;
31 }
32 sum = std::sqrt(sum);
33 }
34
35 // Creating a normalized sparse matrix.
36 Eigen::SparseMatrix<double> mat(which.size(), which.size());
37 {
38 std::vector<int> sizes(which.size());
39 for (size_t c = 0; c < which.size(); ++c) {
40 const auto& current = edges[which[c]];
41 sizes[c] = current.size();
42 }
43 mat.reserve(sizes);
44 }
45
46 for (size_t c = 0; c < which.size(); ++c) {
47 const auto& current = edges[which[c]];
48 for (const auto& f : current) {
49 auto new_index = indices[f.first];
50 if (c < new_index) { // upper-triangular only.
51 break;
52 }
53 mat.insert(new_index, c) = -f.second / sums[new_index] / sums[c];
54 }
55 mat.insert(c, c) = 1;
56 }
57 mat.makeCompressed();
58
59 // Finding the smallest eigenvalues & their eigenvectors,
60 // using the shift-and-invert mode as recommended.
61 const int nobs = mat.rows();
62 int nev = std::min(ndim + 1, nobs); // +1 from uwot:::normalized_laplacian_init
63 int ncv = std::min(nobs, std::max(2 * nev, 20)); // from RSpectra:::eigs_real_sym. I don't make the rules.
64
65 Spectra::SparseSymMatProd<double, Eigen::Upper> op(mat);
66 Spectra::SymEigsSolver<typename std::remove_reference<decltype(op)>::type> eigs(op, nev, ncv);
67
68 eigs.init();
69 eigs.compute(Spectra::SortRule::SmallestMagn);
70
71// Spectra::SparseSymShiftSolve<double, Eigen::Upper> op(mat);
72// Spectra::SymEigsShiftSolver<Spectra::SparseSymShiftSolve<double, Eigen::Upper> > eigs(op, nev, ncv, 0.0);
73//
74// eigs.init();
75// eigs.compute(Spectra::SortRule::LargestMagn);
76
77 if (eigs.info() == Spectra::CompInfo::Successful) {
78 return eigs.eigenvectors().leftCols(ndim);
79
80 } else {
81 size_t order = which.size();
82
83 // Falling back to random initialization.
84 Eigen::MatrixXd output(order, ndim);
85 std::mt19937_64 rng(1234567890);
86 std::uniform_real_distribution<> dist(0, 1);
87
88 auto outptr = output.data();
89 for (size_t i = 0; i < order * ndim; ++i) {
90 outptr[i] = dist(rng);
91 }
92 return output;
93 }
94}

96inline void spectral_init(const NeighborList& edges, int ndim, double * vals) {
97 auto mapping = find_components(edges);
98
99 for (size_t c = 0; c < mapping.ncomponents(); ++c) {
100 auto eigs = normalized_laplacian_by_component(edges, mapping, c, ndim);
101 const auto& reversed = mapping.reversed[c];
102
103 // Getting the maximum value; this is assumed to be non-zero,
104 // otherwise this entire thing is futile.
105 const double max_val = std::max(std::abs(eigs.minCoeff()), std::abs(eigs.maxCoeff()));
106 const double expansion = (max_val > 0 ? 10 / max_val : 1);
107
108 for (size_t r = 0; r < reversed.size(); ++r) {
109 size_t offset = reversed[r] * ndim;
110 for (int d = 0; d < ndim; ++d) {
111 vals[offset + d] = eigs(r, d) * expansion; // TODO: put back the jitter step.
112 }
113 }
114 }
115}
