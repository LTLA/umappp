19template<typename Float>
20bool normalized_laplacian(const NeighborList<Float>& edges, int ndim, Float* Y) {
21 std::vector<double> sums(edges.size());
22 std::vector<int> sizes(edges.size());
23
24 for (size_t c = 0; c < edges.size(); ++c) {
25 const auto& current = edges[c];
26 sizes[c] = current.size() + 1; // +1 for self, assuming that no entry of 'current' is equal to 'c'.
27
28 double& sum = sums[c];
29 for (const auto& f : current) {
30 sum += f.second;
31 }
32 sum = std::sqrt(sum);
33 }
34
35 // Creating a normalized sparse matrix.
36 Eigen::SparseMatrix<double> mat(edges.size(), edges.size());
37 mat.reserve(sizes);
38
39 for (size_t c = 0; c < edges.size(); ++c) {
40 const auto& current = edges[c];
41 size_t i = 0;
42
43 for (; i < current.size() && current[i].first < c; ++i) {
44 const auto& f = current[i];
45 mat.insert(f.first, c) = -f.second / sums[f.first] / sums[c];
46 }
47
48 mat.insert(c, c) = 1;
49
50 for (; i < current.size(); ++i) {
51 const auto& f = current[i];
52 mat.insert(f.first, c) = -f.second / sums[f.first] / sums[c];
53 }
54 }
55 mat.makeCompressed();
56
57 /* We want to find the eigenvectors corresponding to the 'ndim' smallest
58 * positive eigenvalues, as these define a nice initial partition of the
59 * observations (i.e., weak-to-no edges = small eigenvalues). Unfortunately,
60 * the best algorithms are designed to find the largest eigenvalues/vectors.
61 *
62 * So, we observe that the normalized laplacian is positive semi-definite
63 * where the smallest eigenvalue is zero and the largest _possible_
64 * eigenvalue is 2. Thus, we shift the matrix (i.e., '2 * I - L') and then
65 * finding the largest 'ndim + 1' eigenvalues from the shifted matrix.
66 * These correspond to the smallest 'ndim + 1' eigenvalues from the
67 * original matrix. This is obvious when we realize that the eigenvectors
68 * of A are the same as the eigenvectors of (xI - A), but the order of
69 * eigenvalues is reversed because of the negation.
70 *
71 * Initially motivated by comments at yixuan/spectra#126 but I misread the
72 * equations so this approach (while correct) is not what is described in
73 * those links. Also thanks to jlmelville for the max eigenvalue hint,
74 * see LTLA/umappp#4 for the discussion.
75 */
76 mat *= -1;
77 mat.diagonal().array() += static_cast<double>(2);
78
79 irlba::Irlba runner;
80 auto actual = runner.set_number(ndim + 1).run(mat);
81 auto ev = actual.U.rightCols(ndim);
82
83 // Getting the maximum value; this is assumed to be non-zero,
84 // otherwise this entire thing is futile.
85 const double max_val = std::max(std::abs(ev.minCoeff()), std::abs(ev.maxCoeff()));
86 const double expansion = (max_val > 0 ? 10 / max_val : 1);
87
88 for (size_t c = 0; c < edges.size(); ++c) {
89 size_t offset = c * ndim;
90 for (int d = 0; d < ndim; ++d) {
91 Y[offset + d] = ev(c, d) * expansion; // TODO: put back the jitter step.
92 }
93 }
94 return true;
95}

97template<typename Float>
98bool has_multiple_components(const NeighborList<Float>& edges) {
99 if (!edges.size()) {
100 return false;
101 }
102
103 size_t in_component = 1;
104 std::vector<int> remaining(1, 0);
105 std::vector<int> mapping(edges.size(), -1);
106 mapping[0] = 0;
107
108 do {
109 int curfriend = remaining.back();
110 remaining.pop_back();
111
112 for (const auto& ff : edges[curfriend]) {
113 if (mapping[ff.first] == -1) {
114 remaining.push_back(ff.first);
115 mapping[ff.first] = 0;
116 ++in_component;
117 }
118 }
119 } while (remaining.size());
120
121 return in_component != edges.size();
122}

124template<typename Float>
125bool spectral_init(const NeighborList<Float>& edges, int ndim, Float* vals) {
126 if (!has_multiple_components(edges)) {
127 if (normalized_laplacian(edges, ndim, vals)) {
128 return true;
129 }
130 }
131 return false;
132}

137 for (size_t i = 0; i < nobs * ndim; ++i) {
