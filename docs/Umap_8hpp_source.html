397 Status initialize(NeighborList<Float> x, int ndim, Float* embedding) const {
398 neighbor_similarities(x, local_connectivity, bandwidth);
399 combine_neighbor_sets(x, mix_ratio);
400
401 // Choosing the manner of initialization.
402 if (init == SPECTRAL || init == SPECTRAL_ONLY) {
403 bool attempt = spectral_init(x, ndim, embedding);
404 if (!attempt && init == SPECTRAL) {
405 random_init(x.size(), ndim, embedding);
406 }
407 } else if (init == RANDOM) {
408 random_init(x.size(), ndim, embedding);
409 }
410
411 // Finding a good a/b pair.
412 Float a_ = a;
413 Float b_ = b;
414 if (a_ <= 0 || b_ <= 0) {
415 auto found = find_ab(spread, min_dist);
416 a_ = found.first;
417 b_ = found.second;
418 }
419
420 return Status(
421 similarities_to_epochs(x, num_epochs, negative_sample_rate),
422 seed,
423 a_,
424 b_
425 );
426 }

550 template<class Algorithm>
551 Status initialize(const Algorithm* searcher, int ndim, Float* embedding) {
552 const size_t N = searcher->nobs();
553
554#ifdef _OPENMP
555 NeighborList<Float> output(N);
556#else
557 NeighborList<Float> output;
558 output.reserve(N);
559#endif
560
561 #pragma omp parallel for
562 for (size_t i = 0; i < N; ++i) {
563#ifdef _OPENMP
564 output[i] = searcher->find_nearest_neighbors(i, num_neighbors);
565#else
566 output.emplace_back(searcher->find_nearest_neighbors(i, num_neighbors));
567#endif
568 }
569
570 return initialize(output, ndim, embedding);
571 }
