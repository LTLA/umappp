386 Status initialize(NeighborList x, int ndim, double* embedding) const {
387 neighbor_similarities(x, local_connectivity, bandwidth);
388 combine_neighbor_sets(x, mix_ratio);
389
390 // Choosing the manner of initialization.
391 if (init == SPECTRAL || init == SPECTRAL_ONLY) {
392 bool attempt = spectral_init(x, ndim, embedding);
393 if (!attempt && init == SPECTRAL) {
394 random_init(x.size(), ndim, embedding);
395 }
396 } else if (init == RANDOM) {
397 random_init(x.size(), ndim, embedding);
398 }
399
400 // Finding a good a/b pair.
401 double a_ = a;
402 double b_ = b;
403 if (a_ <= 0 || b_ <= 0) {
404 auto found = find_ab(spread, min_dist);
405 a_ = found.first;
406 b_ = found.second;
407 }
408
409 return Status(
410 similarities_to_epochs(x, num_epochs, negative_sample_rate),
411 seed,
412 a_,
413 b_
414 );
415 }

539 template<class Algorithm>
540 Status initialize(const Algorithm* searcher, int ndim, double* embedding) {
541 const size_t N = searcher->nobs();
542
543#ifdef _OPENMP
544 NeighborList output(N);
545#else
546 NeighborList output;
547 output.reserve(N);
548#endif
549
550 #pragma omp parallel for
551 for (size_t i = 0; i < N; ++i) {
552#ifdef _OPENMP
553 output[i] = searcher->find_nearest_neighbors(i, num_neighbors);
554#else
555 output.emplace_back(searcher->find_nearest_neighbors(i, num_neighbors));
556#endif
557 }
558
559 return initialize(output, ndim, embedding);
560 }
