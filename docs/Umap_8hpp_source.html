374 Status initialize(NeighborList x, int ndim, double* embedding) const {
375 neighbor_similarities(x, local_connectivity, bandwidth);
376 combine_neighbor_sets(x, mix_ratio);
377
378 // Running spectral initialization.
379 if (init) {
380 spectral_init(x, ndim, embedding);
381 }
382
383 // Finding a good a/b pair.
384 double a_ = a;
385 double b_ = b;
386 if (a_ <= 0 || b_ <= 0) {
387 auto found = find_ab(spread, min_dist);
388 a_ = found.first;
389 b_ = found.second;
390 }
391
392 return Status(
393 similarities_to_epochs(x, num_epochs, negative_sample_rate),
394 seed,
395 a_,
396 b_
397 );
398 }

522 template<class Algorithm>
523 Status initialize(const Algorithm* searcher, int ndim, double* embedding) {
524 const size_t N = searcher->nobs();
525
526#ifdef _OPENMP
527 NeighborList output(N);
528#else
529 NeighborList output;
530 output.reserve(N);
531#endif
532
533 #pragma omp parallel for
534 for (size_t i = 0; i < N; ++i) {
535#ifdef _OPENMP
536 output[i] = searcher->find_nearest_neighbors(i, num_neighbors);
537#else
538 output.emplace_back(searcher->find_nearest_neighbors(i, num_neighbors));
539#endif
540 }
541
542 return initialize(output, ndim, embedding);
543 }
