343 Status initialize(NeighborList x, int ndim, double* embedding) const {
344 neighbor_similarities(x, local_connectivity, bandwidth);
345 combine_neighbor_sets(x, mix_ratio);
346
347 // Running spectral initialization.
348 if (init) {
349 spectral_init(x, ndim, embedding);

351
352 // Finding a good a/b pair.
353 double a_ = a;
354 double b_ = b;
355 if (a_ <= 0 || b_ <= 0) {
356 auto found = find_ab(spread, min_dist);
357 a_ = found.first;
358 b_ = found.second;
359 }
360
361 return Status(
362 similarities_to_epochs(x, num_epochs, negative_sample_rate),
363 seed,
364 a_,
365 b_
366 );
367 }

475 template<class Algorithm>
476 Status initialize(const Algorithm* searcher, int ndim, double* embedding) {
477 NeighborList output;
478 const size_t N = searcher->nobs();
479 output.reserve(N);
480
481 #pragma omp parallel for
482 for (size_t i = 0; i < N; ++i) {
483 auto out = searcher->find_nearest_neighbors(i, num_neighbors);
484 output.emplace_back(std::move(out));
485 }
486
487 return initialize(output, ndim, embedding);
488 }
