449 Status initialize(NeighborList<Float> x, int ndim, Float* embedding) const {
450 neighbor_similarities(x, local_connectivity, bandwidth);
451 combine_neighbor_sets(x, mix_ratio);
452
453 // Choosing the manner of initialization.
454 if (init == SPECTRAL || init == SPECTRAL_ONLY) {
455 bool attempt = spectral_init(x, ndim, embedding);
456 if (!attempt && init == SPECTRAL) {
457 random_init(x.size(), ndim, embedding);
458 }
459 } else if (init == RANDOM) {
460 random_init(x.size(), ndim, embedding);
461 }
462
463 // Finding a good a/b pair.
464 auto pcopy = rparams;
465 if (pcopy.a <= 0 || pcopy.b <= 0) {
466 auto found = find_ab(spread, min_dist);
467 pcopy.a = found.first;
468 pcopy.b = found.second;
469 }
470
471 return Status(
472 similarities_to_epochs(x, num_epochs, negative_sample_rate),
473 seed,
474 std::move(pcopy)
475 );
476 }
